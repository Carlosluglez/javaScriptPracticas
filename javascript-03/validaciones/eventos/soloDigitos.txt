document.formulario.telefono.onkeypress = soloDigitos;

function soloDigitos(elevento) {
    let evento = elevento || window.event;

    let mostrar = true;
    let regExp = new RegExp("^\\d$");
    let caracterPulsado = String.fromCharCode(evento.charCode);

    if (!regExp.test(caracterPulsado)) {
        mostrar = false;
    }
    return mostrar;

}
